CREATE DATABASE KwikEMartDB;
SHOW DATABASES;

USE KwikEMartDB;

CREATE TABLE usuarios (
	nome_usuario VARCHAR(100),
    senha VARCHAR(60)
);

CREATE TABLE produtos ( 
	cod_barras TEXT,
    descricao TEXT,
    quantidade INT,
    preco DOUBLE
);

CREATE TABLE vendas ( 
	total_compra DOUBLE,
    total_pagamento DOUBLE,
    troco DOUBLE,
    forma_pagamento TEXT
);

CREATE TABLE compras (
	total_compra DOUBLE
);
